linux/unix/macos
oc create secret generic mosquitto-certs \
  --from-file=server.crt \
  --from-file=server.key \
  --from-file=ca.crt

windows powershell:
oc create secret generic mosquitto-certs `
  --from-file=server.crt="$(Get-Location)\certs\server.crt" `
  --from-file=server.key="$(Get-Location)\certs\server.key" `
  --from-file=ca.crt="$(Get-Location)\certs\ca.crt"


gitbash:
oc create secret generic mosquitto-certs \
  --from-file=certs/server.crt \
  --from-file=certs/server.key \
  --from-file=certs/ca.crt
